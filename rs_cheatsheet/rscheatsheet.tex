% --- Small TikZ helper: bordered filled outer box with dashed inner non-filled boxes in a row
% Usage: \tikzboxrow{<size>}{<color>}{<n>} where
%  - <size> is a TeX length (e.g. 20mm, 1.2cm, 50pt) for the outer square side,
%  - <color> is any color understood by xcolor (e.g. orange!60, blue, red!30),
%  - <n> is the number of inner dashed, non-filled boxes to place in a row.
% The inner boxes are automatically scaled to fit with a small padding and gap.
\newlength{\memlouterboxheight}
\newlength{\memlouterboxwidth}
\newlength{\memlinnerboxpad}
\newlength{\memlinnerboxeswidth}
\newlength{\memlinnerboxesheight}
\newlength{\memlinnerboxesgap}
\newlength{\memlinnerboxwidth}
\newlength{\memlinnerboxXshift}

\newcommand{\memlboxrow}[4]{%
  % #1 = width, #2 = height, #3 = color, #4 = n
  \begingroup
  \setlength{\memlouterboxwidth}{#1} % outer square size
  \setlength{\memlouterboxheight}{#2} % outer square size
  % padding, gap and corner radius as percentages of outer size (reasonable defaults)
  \setlength{\memlinnerboxpad}{\dimexpr \memlouterboxheight * 20 / 100\relax}% 20%
  \setlength{\memlinnerboxesgap}{\dimexpr \memlouterboxheight * 10 / 100\relax}% 10%
  \setlength{\memlinnerboxeswidth}{\dimexpr \memlouterboxwidth - 2\memlinnerboxpad\relax}% inner available width
  \setlength{\memlinnerboxesheight}{\dimexpr \memlouterboxheight - 2\memlinnerboxpad\relax}% keep square inner area
  % compute inner box width as a TeX length using PGF math; pass explicit numeric lengths
  % via "\\the\\<length>" so PGF gets numeric values it can evaluate
  \pgfmathsetlength{\memlinnerboxwidth}{(\the\memlinnerboxeswidth - ((#4 - 1)*\the\memlinnerboxesgap))/#4}
    \begin{tikzpicture}[inner sep=+0pt, outer sep=+0pt, baseline={([yshift=-0.6ex]current bounding box.center)}]

        % outer box: filled, slightly rounded, border chosen darker than fill
        \node[draw=#3!black, fill=#3, rounded corners=2pt, 
            minimum width=\memlouterboxwidth, minimum height=\memlouterboxheight, inner sep=0pt] (outer) {};

        % place inner dashed, non-filled boxes centered vertically inside outer
        \foreach \i in {1,...,#4} {
          % compute horizontal center position of current inner box (as a TeX length)
          % use "\\the\\<length>" so PGF math sees numeric values with units
          \pgfmathsetlength{\memlinnerboxXshift}{\the\memlinnerboxpad + ((\i-1)*(\the\memlinnerboxwidth + \the\memlinnerboxesgap) + \the\memlinnerboxwidth/2)}


          \node[draw=#3!black, solid, fill=#3!gray, rounded corners=0.5pt, line width=0.5pt,
                minimum width=\memlinnerboxwidth, minimum height=\memlinnerboxesheight, inner sep=0pt]
                at ($(outer.west)+ (\memlinnerboxXshift,0)$) {};
        }
    \end{tikzpicture}%
  \endgroup
}

  % --- Overview -----------------------------------------------------------
  \begin{myboxed}{Overview}
        {
            Rust is a general-purpose programming language focused on performance, memory safety without a garbage collector, and concurrency, using an ownership system to prevent memory errors.
            \vspace{0.5ex}
            \newline 
            \href{https://cheats.rs/\#hello-rust}{\textbf{Strengths}}:
            \vspace{0.5ex}
            \begin{itemize}[nosep,leftmargin=2em]
                \item Compiled code about same performance as C / C++, and excellent memory and energy efficiency.
                \item Can avoid 70\% of all safety issues present in C / C++, and most memory issues.
                \item Strong type system prevents data races, brings 'fearless concurrency' (amongst others).
                \item Seamless C interop, and dozens of supported platforms (based on LLVM).
            \end{itemize} 
            \vspace{1ex}
            \href{https://rustup.rs/}{\textbf{Installation}}:
            \vspace{-1ex}%
            \begin{minted}[]{bash}
                curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh
            \end{minted}

        }
    \end{myboxed}%
    \vspace{-2ex}%

    \vspace{\fill}

    % --- Toolbox -----------------------------------------------------------
    \begin{myboxed}{ Toolbox \hfill }
        {
            \textbf{Rustup} is used to install and manage Rust toolchains. Toolchains are complete installations of Rust compiler and tools. \vfill
            \vspace{0.5ex}
            \begin{tabular}{@{}p{0.48\columnwidth}p{0.49\columnwidth}@{}}
                \texttt{rustup show}                & Show installed toolchains                     \\
                \texttt{rustup update}              & Update all toolchains                         \\
                \texttt{rustup default TC}   & Set the default toolchain to \texttt{TC}             \\
                \texttt{rustup component list}      & List available components                     \\
                \texttt{rustup component add NAME}  & Add component \texttt{NAME}                   \\
                \texttt{rustup target list}         & List available compilation targets            \\
                \texttt{rustup target add NAME}     & Add a compilation target \texttt{NAME}        \\
                \texttt{rustup docs}                & Open local Rust documentation in standard web browser   \\
                \texttt{rustup uninstall TC}        & Uninstall toolchain \texttt{TC}               \\ 
            \end{tabular}

            \smallskip

            \href{https://rust-lang.github.io/rustup/concepts/components.html}{\textbf{Components}}: parts of toolchains, partly \textit{man}datory partly \textit{opt}ional. \vfill
            \vspace{0.5ex}
            \begin{tabular}{@{}p{0.25\columnwidth}p{0.07\columnwidth}p{0.63\columnwidth}@{}}
                \texttt{rustc}          & \textit{man} & The Rust compiler and \texttt{Rustdoc} \\
                \texttt{cargo}          & \textit{man} & Package manager and build tool  \\
                \texttt{rustfmt}        & \textit{opt} & Code formatter  \\
                \texttt{rust-std}       & \textit{man} & Rust standard library, each target separate  \\
                \texttt{rust-docs}      & \textit{opt} & Local copy of the Rust documentation  \\
                \texttt{rust-analyzer}  & \textit{opt} & Language server for editors and IDEs  \\
                \texttt{clippy}         & \textit{opt} & Lint tool providing extra checks for common mistakes and stylistic choices  \\
                \texttt{miri}           & \textit{opt} & Experimental Rust interpreter \\
                \texttt{rust-src}       & \textit{opt} & Local copy of standard library source  \\
                \texttt{rust-mingw}     & \textit{opt} & Linker and platform libraries for building on the \textit{x86\_64-pc-windows-gnu} platform  \\
                \texttt{llvm-tools}     & \textit{opt} & Collection of LLVM tools. Unstable \\
                \texttt{rustc-dev}      & \textit{opt} & Compiler as a library, only needed for development of tools that link to the compiler  \\
            \end{tabular}

            \smallskip

            \textbf{Cargo} is used to build and run Rust projects. \vfill
            \vspace{0.5ex}
            \begin{tabular}{@{}p{0.46\columnwidth}p{0.51\columnwidth}@{}}
                \texttt{cargo init}                & Create a new binary project                                   \\
                \texttt{cargo init --lib}          & Create a new library project                                  \\
                \texttt{cargo check}               & Check code for errors                                         \\
                \texttt{cargo clippy}              & Lint code                                                     \\
                \texttt{cargo doc}                 & Generate project documentation                                        \\
                \texttt{cargo run}                 & Run the project                                               \\
                \texttt{cargo run --bin NAME}      & Run a specific project binary                                 \\
                \texttt{cargo build}               & Build everything in debug mode                                \\
                \texttt{cargo build --bin NAME}    & Build binary \texttt{NAME} in debug mode                               \\
                \texttt{cargo build --release}     & Build everything in release mode                              \\
                \texttt{cargo build --target NAME} & Build for a specific target \texttt{NAME}                                  \\
                \texttt{cargo --explain CODE}      & Details on compiler error \texttt{CODE}                                \\
                \texttt{cargo test}                & Run all tests                                                 \\
                \texttt{cargo test TEST\_NAME}     & Run a specific test \texttt{TEST\_NAME}                                          \\
                \texttt{cargo test --doc}          & Run doctests only                                             \\
                \texttt{cargo test --examples}     & Run tests for example code only                               \\
                \texttt{cargo bench}               & Run benchmarks
            \end{tabular}
        }
    \end{myboxed}%
